  <script>
  const buttons = document.querySelectorAll('.date-btn');
  const cards = document.querySelectorAll('.trip-card');

  function filterTrips(month) {
    cards.forEach(card => {
      const cardMonth = card.getAttribute('data-month');
      card.style.display = cardMonth === month ? 'block' : 'none';
    });
  }

  // Initial load: filter by the active button's month
  const defaultActive = document.querySelector('.date-btn.active');
  if (defaultActive) {
    filterTrips(defaultActive.getAttribute('data-month'));
  }

  // Add click event listeners
  buttons.forEach(button => {
    button.addEventListener('click', () => {
      buttons.forEach(btn => btn.classList.remove('active'));
      button.classList.add('active');
      filterTrips(button.getAttribute('data-month'));
    });
  });
</script>
